В результате выполнения следующей программы:
<code>using System;
class Program {
	delegate int Del(int n);
	static int Method(Del f) {
		int n = 1;
		return f(n);
	}

	static void Main() {
		int x = 1, i = 0;
		while (i++ < 3) {
			Console.Write(Method((j) => { return j + x + i; }));
		}
	}
}</code>
